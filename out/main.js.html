/**
*@global
 */
var field = document.getElementById('field'),
    table = document.getElementById('table');

/**
 *@event 
 *@description get number || dot || operation
 * and add it to input field
 */
table.onclick = function(e){
  var target = e.target;
  
  if(target.closest('.number') || target.closest('.dot')){
    field.value += target.innerHTML;  
  }

  if(target.closest('.operation')){
    field.value += ' ';
    field.value += target.innerHTML;
    field.value += ' ';
  }

  if(target.id == 'clear'){
    clear();
  }
  if(target.id == 'calculate'){
    calc();
  }

}

/*
 *@function
 *@name clear
 *@description clear input.value 
 */
function clear(){
  field.value = '';
}
/**
 *@function
 *@name calc
 *@description split string from #field into firstOperand, operator,
 *and secondOperand
 *check operator by switch construction
 *and make operation and add result to #field
 */

function calc(){
  var expression = field.value.split(' '),
    a = +expression[0],
    b = +expression[2],
    operator = expression[1],
    result;

  switch (operator){
    case '+':
      result = (a+b);
      break;
    case '-':
      result = (a-b);
      break;
    case '*':
      result = (a*b);
      break;
    case '/':
      result = (a/b);
      break;
    default: result+= "null";
    break;
  }

  field.value = result;
}
